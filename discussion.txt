🎨 The Retro Tour de France Style — What Defines It
This style has a very specific visual DNA. If you follow these rules, your AI outputs will look like real vintage cycling posters.

1. Bold, flat color blocks
No gradients. No soft shading.
Think:

Mustard yellow

Brick red

Navy blue

Forest green

Cream / off‑white

2. Grainy print texture
Old posters were printed with imperfect ink.
You want:

Slight grain

Faded edges

Subtle halftone dots

3. Simplified shapes
Mountains become triangles.
Trees become geometric silhouettes.
Cyclists are stylized, not realistic.

4. Dramatic compositions
Classic motifs:

A lone cyclist climbing a huge mountain

A winding road disappearing into the distance

A peloton in silhouette

A cyclist in profile with oversized mountains behind

5. Vintage typography (optional)
Big blocky letters like:

“COL DU GALIBIER”

“MONT VENTOUX”

“LE TOUR”

“GRAVEL FINLAND”

Typography is optional — you can add it later in Figma or Canva.

Python batch‑generation script (retro Tour de France cycling posters)
Below is a self‑contained Python script that:

Builds prompts from a template

Sends them to a generic image generation API

Saves the returned images to disk

python
import os
import itertools
import time
import requests

# ==========================
# CONFIG
# ==========================

API_URL = "https://your-image-api-endpoint.com/v1/images/generations"
API_KEY = "YOUR_API_KEY_HERE"

OUTPUT_DIR = "retro_tdf_posters"
os.makedirs(OUTPUT_DIR, exist_ok=True)

# Limit total images per run so you don't accidentally generate thousands
MAX_IMAGES = 50

# ==========================
# VARIATION LISTS
# ==========================

cyclist_types = [
    "road cyclist",
    "gravel cyclist",
    "vintage steel bike rider",
    "peloton",
    "solo climber",
    "breakaway rider",
    "sprinter",
    "time trialist",
    "MTB rider",
    "silhouette cyclist",
]

environments = [
    "Col du Galibier",
    "Mont Ventoux",
    "the Pyrenees",
    "a Finnish pine forest",
    "a lakeside road",
    "an alpine village",
    "coastal cliffs",
    "rolling hills",
    "lavender fields",
    "a snowy forest",
    "a gravel road",
    "a mountain pass",
    "countryside",
    "city streets",
    "a desert canyon",
]

color_palettes = [
    "warm retro colors (yellow, orange, red)",
    "cool retro colors (blue, green, cream)",
    "faded pastel retro tones",
    "high-contrast vintage colors",
    "muted earth tones",
]

compositions = [
    "diagonal climb composition",
    "winding road composition",
    "centered silhouette composition",
    "distant rider composition",
    "close-up profile composition",
]

# ==========================
# PROMPT TEMPLATE
# ==========================

def build_prompt(cyclist_type, environment, palette, composition):
    return (
        f"Retro 1960s Tour de France cycling poster of a {cyclist_type} "
        f"riding through {environment}. {composition}, "
        f"{palette}, bold flat colors, simplified geometric shapes, "
        f"grainy vintage print texture, muted retro palette, dramatic composition, high resolution."
    )

# ==========================
# GENERIC IMAGE API CALL
# ==========================

def generate_image(prompt, index):
    """
    Adjust this function to match your actual image API.
    Example assumes a JSON POST returning base64 or URL.
    """
    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json",
    }

    payload = {
        "prompt": prompt,
        "n": 1,
        "size": "1024x1024",  # adjust as needed
    }

    response = requests.post(API_URL, json=payload, headers=headers)
    response.raise_for_status()
    data = response.json()

    # Adapt this part depending on your API's response format.
    # Example: data["images"][0]["b64_json"] or data["data"][0]["url"]
    # Here we'll assume a URL is returned.
    image_url = data["data"][0]["url"]

    # Download the image
    img_resp = requests.get(image_url)
    img_resp.raise_for_status()

    filename = os.path.join(OUTPUT_DIR, f"retro_tdf_{index:03d}.png")
    with open(filename, "wb") as f:
        f.write(img_resp.content)

    print(f"Saved: {filename}")
    return filename

# ==========================
# MAIN BATCH GENERATION
# ==========================

def main():
    combos = itertools.product(cyclist_types, environments, color_palettes, compositions)

    count = 0
    for cyclist_type, environment, palette, composition in combos:
        if count >= MAX_IMAGES:
            break

        prompt = build_prompt(cyclist_type, environment, palette, composition)
        print(f"\n[{count+1}/{MAX_IMAGES}] Prompt:\n{prompt}\n")

        try:
            generate_image(prompt, count)
        except Exception as e:
            print(f"Error generating image {count}: {e}")

        count += 1
        time.sleep(1)  # small delay to avoid rate limits

    print(f"\nDone. Generated {count} images into '{OUTPUT_DIR}'.")

if __name__ == "__main__":
    main()

import base64
import requests

API_KEY = "YOUR_OPENAI_API_KEY"

def generate_image(prompt):
    url = "https://api.openai.com/v1/images/generations"
    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json"
    }
    payload = {
        "model": "gpt-image-1",
        "prompt": prompt,
        "size": "1024x1024"
    }

    response = requests.post(url, json=payload, headers=headers)
    response.raise_for_status()
    data = response.json()

    # The API returns base64-encoded image data
    image_b64 = data["data"][0]["b64_json"]
    image_bytes = base64.b64decode(image_b64)

    return image_bytes
